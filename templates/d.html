import paho.mqtt.client as mqtt # The callback for when the client receives a CONNACK response from the server. def on_connect(client, userdata, flags, rc): print("Connected with result code " + str(rc)) client.subscribe("Temp/Air/Light") # The callback for when a PUBLISH message is received from the server. def on_message(client, userdata, msg): print(msg.topic +" " + str(msg.payload)) client = mqtt.Client() client.on_connect = on_connect client.on_message = on_message client.connect("52.230.126.225") client.loop_forever()
